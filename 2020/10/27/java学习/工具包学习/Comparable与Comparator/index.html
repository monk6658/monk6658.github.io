    import java.util.Objects;

/**
 * е®һзҺ°ComparatorжҺҘеҸЈзҡ„зұ»еҸҜд»Ҙж–№дҫҝзҡ„жҺ’еәҸпјҢ иҰҶеҶҷcompareToжҺҘеҸЈ
 * @author zxl
 * @date 2020/10/27 18:00
 */
public class Student implements Comparable<Student>{

    /*** еҗҚеӯ— */
    private String name;

    /*** е№ҙйҫ„ */
    private int age;

    /*** еҲҶж•° */
    private float score;

    /**
     * жҺ’еәҸи§„еҲҷпјҡд»ҘеҲҶж•°йҷҚеәҸжҺ’еәҸпјҢиӢҘеҲҶж•°дёҖиҮҙпјҢеҲҷж №жҚ®е№ҙйҫ„еҚҮеәҸжҺ’еәҸ
     * @param o жҜ”иҫғеҸӮж•°
     * @return жҜ”иҫғз»“жһң -1еӨ§дәҺ   0зӯүдәҺ   1е°ҸдәҺ
     */
    @Override
    public int compareTo(Student o) {
        /*
         * this.age - o.age еҚҮеәҸ
         * o.score - this.score йҷҚеәҸ
         */
        return Objects.equals(o.score,this.score) ? this.age - o.age : (int) (o.score - this.score);
    }

    public Student(String name, int age, float score) {
        this.name = name;
        this.age = age;
        this.score = score;
    }

    @Override
    public String toString() {
        return "Student{" +
                "name='" + name + '\'' +
                ", age=" + age +
                ", score=" + score +
                '}';
    }
}
